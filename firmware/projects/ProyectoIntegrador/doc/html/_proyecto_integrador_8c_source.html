186void read_data(uint8_t *data, uint8_t length)
187{
188 // Si el boton esta en On ('O') el robot debe encenderse
189 // Si el boton esta en Off ('o') el robot debe apagarse
190 switch (data[0])
191 {
192 case 'O':
193 control = true;
194 BleSendString("*ERobot encendido\n*"); // Envio mensaje a la app del celular
195 break;
196 case 'o':
197 control = false;
198 BleSendString("*ERobot apagado\n*"); // Envio mensaje a la app del celular
199 break;
200 default:
201 break;
202 }

209void controlRuedas(char dir)
210{
211 switch (dir)
212 {
213 case 'A': // avanzar
214 GPIOOn(MIB);
215 GPIOOn(MDB);
216
217 PWMOn(pwm_motores[0]);
218 PWMOn(pwm_motores[1]);
219
220 // Velocidad
221 PWMSetDutyCycle(pwm_motores[0], 60);
222 PWMSetDutyCycle(pwm_motores[1], 60);
223 break;
224
225 case 'F': // frenar
226 GPIOOn(MIB);
227 GPIOOn(MDB);
228
229 PWMSetDutyCycle(pwm_motores[0], 100);
230 PWMSetDutyCycle(pwm_motores[1], 100);
231 break;
232
233 case 'D': // giro derecha
234 GPIOOn(MIB);
235 GPIOOn(MDB);
236
237 PWMOn(pwm_motores[0]);
238 PWMOn(pwm_motores[1]);
239
240 // velocidad
241 PWMSetDutyCycle(pwm_motores[0], 35); // rueda izq
242 PWMSetDutyCycle(pwm_motores[1], 65); // rueda der
243 break;
244
245 case 'I': // giro izquierda
246 GPIOOn(MIB);
247 GPIOOn(MDB);
248
249 PWMOn(pwm_motores[0]);
250 PWMOn(pwm_motores[1]);

254 PWMSetDutyCycle(pwm_motores[1], 35); // rueda der
255 break;
256
257 default:
258 break;
259 }

265void controlDistancia(void *pvParameter)
266{
267
268 while (1)
269 {
270 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
271 if (control)
272 {
273 distancia = HcSr04ReadDistanceInCentimeters();
274
275 static uint16_t distDer, distIzq;
276
277 if (distancia > 15)
278 {
279 BleSendString("*EAvanzando\n*");
280 controlRuedas('A');
281 }
282 else if (distancia <= 15)
283 {
284 // apago motores
285 BleSendString("*EFreno\n*");
286 controlRuedas('F');
287
288 ServoMove(SERVO_2, giroDer); // se queda en esta posicion
289 vTaskDelay(850 / portTICK_PERIOD_MS);
290 distDer = HcSr04ReadDistanceInCentimeters();
291 ServoMove(SERVO_2, posCentral);
292 vTaskDelay(600 / portTICK_PERIOD_MS);
293 ServoMove(SERVO_2, giroIzq);
294 vTaskDelay(850 / portTICK_PERIOD_MS);
295 distIzq = HcSr04ReadDistanceInCentimeters();
296 ServoMove(SERVO_2, posCentral);
297 vTaskDelay(600 / portTICK_PERIOD_MS);
298
299 if (distDer > distIzq)
300 {
301 // grio a la derecha
302 BleSendString("*EGiro a la derecha\n*");
303 controlRuedas('D');
304 vTaskDelay(650 / portTICK_PERIOD_MS);
305 controlRuedas('A');
306 }
307 else if (distDer < distIzq)
308 {
309 // giro a la izquierda
310 BleSendString("*EGiro a la izquierda\n*");
311 controlRuedas('I');
312 vTaskDelay(650 / portTICK_PERIOD_MS);
313 controlRuedas('A');
314 }
315 }
316 }
317 else
318 {
319 controlRuedas('F');
320 }
321 }

327void controlIntensidad(void *pvParameter)
328{
329 while (1)
330 {
331 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
332 if (control)
333 {
334 // El valor leído por LDR_ReadIntensity() será un número entre 0 y 1023
335 // 0 significa máxima resistencia (menos luz) y 1023 mínima resistencia (más luz)
336 intensidad = LDR_ReadIntensity();
337
338 if (intensidad < LUX_NORMAL)
339 {
340 // PRENDO UNA LUZ (LED)
341 GPIOOn(GPIO_LED);
342 }
343 else if (intensidad > LUX_NORMAL)
344 {
345 // APAGO LA LUZ
346 GPIOOff(GPIO_LED);
347 }
348 }
349 }

353void app_main(void)
354{
355 // inicializo hc sr04, servo, LDR y leds
356 HcSr04Init(GPIO_3, GPIO_2);
357 ServoInit(SERVO_2, GPIO_21);
358 LDRInit(CH1);
359 LedsInit();
360
361 // inicializo Bluetooth
362 ble_config_t ble_configuration = {
363 "ROBOT_MICA",
364 read_data};
365 BleInit(&ble_configuration);
366
367 /* Configuracion e inicializacion del timer A */
368 timer_config_t timer_distancia = {
369 .timer = TIMER_A,
370 .period = PERIODO,
371 .func_p = FuncTimerA_Distancia,
372 .param_p = NULL};
373 TimerInit(&timer_distancia);
374
375 /* Configuracion e inicializacion del timer B */
376 timer_config_t timer_oscuridad = {
377 .timer = TIMER_B,
378 .period = PERIODO_LED,
379 .func_p = FuncTimerB_Intesnidad,
380 .param_p = NULL};
381 TimerInit(&timer_oscuridad);
382
383 // Inicializo gpio para el led
384 gpioConf_t LED = {
385 .pin = GPIO_LED,
386 .dir = GPIO_OUTPUT};
387 GPIOInit(LED.pin, LED.dir);
388
389 // gpio puente H y motores
390 gpioConf_t gpio_motores[4];
391
392 // como solo avanza uso dos como pwm
393 gpio_motores[0].pin = MIB;
394 gpio_motores[1].pin = MDB;
395 gpio_motores[2].pin = MIA;
396 gpio_motores[3].pin = MDA;
397
398 // gpio como salida
399 for (int i = 0; i < 4; i++)
400 {
401 gpio_motores[i].dir = GPIO_OUTPUT;
402 }
403
404 // inicializo gpio
405 for (int j = 0; j < 4; j++)
406 {
407 GPIOInit(gpio_motores[j].pin, gpio_motores[j].dir);
408 }
409
410 // pwm
411 pwm_motores[0] = PWM_0;
412 pwm_motores[1] = PWM_1;
413
414 // inicializo pwm
415 PWMInit(pwm_motores[0], MIA, 50);
416 PWMInit(pwm_motores[1], MDA, 50);
417
418 xTaskCreate(&controlDistancia, "controlDistancia", 2048, NULL, 5, &control_distancia_handle);
419 xTaskCreate(&controlIntensidad, "controlIntensidad", 2048, NULL, 5, &control_intensidad_handle);
420
421 TimerStart(timer_distancia.timer);
422 TimerStart(timer_oscuridad.timer);
423
424 // Los leds cambian segun el estado del BLE:
425 // Conectado prende led verde, si se desconecta el led amarillo y si se apaga el led rojo
426 while (1)
427 {
428 vTaskDelay(CONFIG_BLINK_PERIOD / portTICK_PERIOD_MS);
429 switch (BleStatus())
430 {
431 case BLE_OFF:
432 LedOn(LED_3);
433 break;
434 case BLE_DISCONNECTED:
435 LedOff(LED_1);
436 LedToggle(LED_2);
437 break;
438 case BLE_CONNECTED:
439 LedOff(LED_2);
440 LedOn(LED_1);
441 break;
442 }
443 }
444}
