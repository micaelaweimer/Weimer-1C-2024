130void controlPh(void *pvParameter)
131{
132 while (1)
133 {
134 if (control)
135 {
136 float lectura_aux;
137 AnalogInputReadSingle(CH1, &lectura_aux); //leo el dato
138 lectura_aux = lectura_aux*(3/1024); // ajusto segun voltaje maximo
139 lectura = lectura_aux*(14/3); // encuentro el valor entre 0 y 14V
140
141 if (lectura > 6.7)
142 {
143 GPIOOn(bombaPhA.pin); // enciendo bomba pHA
144 GPIOOff(bombaPhB.pin);
145 }
146 else if (lectura < 6.7 | lectura > 6.0)
147 {
148 GPIOOff(bombaPhA.pin); // apago bombas
149 GPIOOff(bombaPhB.pin);
150 }
151 else if (lectura < 6.0)
152 {
153 GPIOOn(bombaPhB.pin); // enciendo bomba pHB
154 GPIOOff(bombaPhA.pin);
155 }
156 }
157 vTaskDelay(PERIODO / portTICK_PERIOD_MS);
158 }
